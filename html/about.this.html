about this（备忘）

1.函数调用

		var x = 1;

		function fn(){

			return this.x;

		}

		fn(); // 1;

this指向函数运行时的作用域

2.作为对象的方法调用

		var x = 1;

		var obj = {
			
			x  : 0,

			fn : function(){

				return this.x;
			
			}

		};

		obj.fn(); // 0;

this指向所在对象

3.构造函数

		var x = 1;

		var Fn = function(){

			this.x = 0;

		};

		var fn = new Fn();

		fn.x  // 0

		x     // 1

3.apply调用模式

		var x = 1;

		function fn(){

			return this.x;

		}

		var obj = {};

		obj.x = 0;

		obj.f = fn;

		obj.f.apply(); // 1

		obj.f.apply(obj); // 0
